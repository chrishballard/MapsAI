    ```typescript
    import { prisma } from '@/lib/prisma';
    import { computeOptimizationScore } from '@/lib/optimization-score';
    import { OptimizationScoreGauge } from './optimization-score-gauge';
    import { AuditCardsGrid } from './audit-cards-grid';
    ```

    - `export async function OptimizationContent({ profileId }: { profileId: string })`
    - Heavy Prisma query:
      ```typescript
      const profile = await prisma.profile.findUnique({
        where: { id: profileId },
        select: {
          id: true,
          name: true,
          reviews: { select: { rating: true, reviewDate: true } },
          posts: { select: { publishedAt: true, status: true } },
          descriptions: {
            select: { id: true, content: true, isApproved: true, isPushed: true, pushedAt: true },
            orderBy: { updatedAt: 'desc' },
            take: 1,
          },
          services: {
            select: { id: true, serviceName: true, description: true, isApproved: true, isPushed: true, pushedAt: true },
          },
        },
      });
      ```
    - If `!profile` return null (shell already handles notFound)
    - Call `computeOptimizationScore(profile)` — returns `{ total, grade, checks }`
    - Sort checks: `const sortedChecks = [...score.checks].sort((a, b) => STATUS_ORDER[a.status] - STATUS_ORDER[b.status])` where `STATUS_ORDER = { critical: 0, warning: 1, good: 2 }`
    - Return JSX:
      - `<div className="space-y-8">`
      - Score section: `<div className="flex items-center gap-8">` containing `<OptimizationScoreGauge total={score.total} grade={score.grade} />` and a text block showing score interpretation
      - Audit section: `<div>` with `<h2 className="text-xl font-semibold text-zinc-900 mb-4">Signal Breakdown</h2>` then `<AuditCardsGrid checks={sortedChecks} />`
      - A placeholder `div` with `id="suggestions-section"` for Plan 03 to populate (or just leave space — Plan 03 adds the suggestions panel)

    IMPORTANT: Call `computeOptimizationScore` in the SERVER component, NOT in a client component. This avoids Date serialization issues across the server/client boundary (Pitfall 1 from research).
